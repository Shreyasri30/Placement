package day_2;


class Bird {
    void sound() {
        System.out.println("Bird makes a sound");
    }
}

class Sparrow extends Bird {

    @Override
    void sound() {
        System.out.println("Sparrow chirps");
    }
}

public class OverridingExample {
    public static void main(String[] args) {

        Bird b = new Sparrow();  
        b.sound();              
    }
}
